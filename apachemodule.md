# apache module 作成手順

### テンプレート作成
```shell
apxs -g -n test
```

### mod_test.c 内容

```c
#include "httpd.h"
#include "http_config.h"
#include "http_protocol.h"
#include "ap_config.h"

/* The sample content handler */
static int test_handler(request_rec *r)
{
    if (strcmp(r->handler, "test")) {　←★handler=="test"の場合のみ処理継続
        return DECLINED;
    }
    r->content_type = "text/html";

    if (!r->header_only)
        ap_rputs("The sample page from mod_test.c\n", r);
    return OK;
}

static void test_register_hooks(apr_pool_t *p)
{
    ap_hook_handler(test_handler, NULL, NULL, APR_HOOK_MIDDLE);
}

/* Dispatch list for API hooks */
module AP_MODULE_DECLARE_DATA test_module = {
    STANDARD20_MODULE_STUFF,
    NULL,                  /* create per-dir    config structures */
    NULL,                  /* merge  per-dir    config structures */
    NULL,                  /* create per-server config structures */
    NULL,                  /* merge  per-server config structures */
    NULL,                  /* table of config file commands       */
    test_register_hooks  /* register hooks                      */
};
```

### コンパイル、インストール
```shell
cd test
apxs -c -i mod_test.c
```

### httpd.conf修正
```apacheconf
LoadModule test_module modules/mod_test.so
<Location "/">
  SetHandler test
</Location>
```

### request_rec構造体の中身をだす

```c
/*
**  mod_recdump.c -- Apache sample recdump module
**  [Autogenerated via ``apxs -n recdump -g'']
**
**  To play with this sample module first compile it into a
**  DSO file and install it into Apache's modules directory
**  by running:
**
**    $ apxs -c -i mod_recdump.c
**
**  Then activate it in Apache's httpd.conf file for instance
**  for the URL /recdump in as follows:
**
**    #   httpd.conf
**    LoadModule recdump_module modules/mod_recdump.so
**    <Location /recdump>
**    SetHandler recdump
**    </Location>
**
**  Then after restarting Apache via
**
**    $ apachectl restart
**
**  you immediately can request the URL /recdump and watch for the
**  output of this module. This can be achieved for instance via:
**
**    $ lynx -mime_header http://localhost/recdump
**
**  The output should be similar to the following one:
**
**    HTTP/1.1 200 OK
**    Date: Tue, 31 Mar 1998 14:42:22 GMT
**    Server: Apache/1.3.4 (Unix)
**    Connection: close
**    Content-Type: text/html
**
**    The sample page from mod_recdump.c
*/

#include "httpd.h"
#include "http_config.h"
#include "http_protocol.h"
#include "ap_config.h"

#define frputs(f, buff, r, prop) \
            sprintf(buff, "%s:" f "</br>\n", #prop, r->prop); \
            ap_rputs(buff, r);
#define crputs(buff, r, prop) frputs("%s", buff, r, prop)
#define irputs(buff, r, prop) frputs("%d", buff, r, prop)

/* The sample content handler */
static int recdump_handler(request_rec *r)
{
    if (strcmp(r->handler, "recdump")) {
        return DECLINED;
    }
    r->content_type = "text/html";

//    if (!r->header_only) {
//        return OK;
//    }

    char buff[4096];

    ap_rputs("request_rec構造体ダンプ</br>\n", r);
    crputs(buff, r, the_request);
    crputs(buff, r, protocol);
    crputs(buff, r, hostname);
    crputs(buff, r, status_line);
    crputs(buff, r, method);
    crputs(buff, r, range);
    crputs(buff, r, content_type);
    crputs(buff, r, handler);
    crputs(buff, r, content_encoding);
    crputs(buff, r, vlist_validator);
    crputs(buff, r, user);
    crputs(buff, r, ap_auth_type);
    crputs(buff, r, unparsed_uri);
    crputs(buff, r, uri);
    crputs(buff, r, filename);
    crputs(buff, r, canonical_filename);
    crputs(buff, r, path_info);
    crputs(buff, r, args);

    irputs(buff, r, assbackwards);
    irputs(buff, r, proxyreq);
    irputs(buff, r, header_only);
    irputs(buff, r, proto_num);
    irputs(buff, r, status);
    irputs(buff, r, method_number);
    irputs(buff, r, chunked);
    irputs(buff, r, read_body);
    irputs(buff, r, read_chunked);
    irputs(buff, r, no_cache);
    irputs(buff, r, no_local_copy);
    irputs(buff, r, used_path_info);
    irputs(buff, r, eos_sent);

    ap_rputs("</br>\n", r);
    ap_rputs("request_rec->connection構造体メンバ</br>\n", r);
    sprintf(buff, "remote-ip:%s<br>\n", r->connection->remote_ip);
    ap_rputs(buff, r);
    sprintf(buff, "remote-host:%s<br>\n", r->connection->remote_host);
    ap_rputs(buff, r);
    sprintf(buff, "remote-logname:%s<br>\n", r->connection->remote_logname);
    ap_rputs(buff, r);
    sprintf(buff, "local_ip:%s<br>\n", r->connection->local_ip);
    ap_rputs(buff, r);
    sprintf(buff, "local_host:%s<br>\n", r->connection->local_host);
    ap_rputs(buff, r);

    //return OK;
    return DECLINED;
}

static void recdump_register_hooks(apr_pool_t *p)
{
    ap_hook_handler(recdump_handler, NULL, NULL, APR_HOOK_MIDDLE);
}

/* Dispatch list for API hooks */
module AP_MODULE_DECLARE_DATA recdump_module = {
    STANDARD20_MODULE_STUFF,
    NULL,                  /* create per-dir    config structures */
    NULL,                  /* merge  per-dir    config structures */
    NULL,                  /* create per-server config structures */
    NULL,                  /* merge  per-server config structures */
    NULL,                  /* table of config file commands       */
    recdump_register_hooks  /* register hooks                      */
};
```
